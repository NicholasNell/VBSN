141 void RadioInit( ) {
142  // Radio initialization
143  RadioEvents.TxDone = OnTxDone;
144  RadioEvents.RxDone = OnRxDone;
145  RadioEvents.TxTimeout = OnTxTimeout;
146  RadioEvents.RxTimeout = OnRxTimeout;
147  RadioEvents.RxError = OnRxError;
148 
149  // detect radio hardware
150  if (Radio.Read(REG_VERSION) == 0x00) {
151  sendUARTpc("Radio could not be detected!\n\r");
152  rfm95Working = false;
153  BoardResetMcu();// if no radio is detected reset the MCU and try again; no radio, no work
154  return;
155  }
156  else {
157  rfm95Working = true;
158  }
159 
160  Radio.Init(&RadioEvents);
161 
162  Radio.SetTxConfig(MODEM_LORA,
163  TX_OUTPUT_POWER,
164  LORA_FREQ_DEV,
165  LORA_BANDWIDTH,
166  LORA_SPREADING_FACTOR,
167  LORA_CODINGRATE,

170  LORA_CRC_ON, LORA_FREQ_HOP_ENABLED, LORA_FREQ_HOP_PERIOD,
171  LORA_IQ_INVERSION_ON, TX_TIMEOUT_VALUE);
172 
173  Radio.SetRxConfig(MODEM_LORA,
174  LORA_BANDWIDTH,
175  LORA_SPREADING_FACTOR,
176  LORA_CODINGRATE, LORA_bandwidthAfc,
177  LORA_PREAMBLE_LENGTH, LORA_SYMBOL_TIMEOUT,

180  LORA_IQ_INVERSION_ON, LORA_RX_CONTINUOUS);
181 
182  Radio.SetMaxPayloadLength(MODEM_LORA, LORA_MAX_PAYLOAD_LEN);
183  Radio.SetPublicNetwork(LORA_IS_PUBLIC_NET, LORA_PRIVATE_SYNCWORD);
184  // set radio frequency channel
185  Radio.SetChannel( RF_FREQUENCY);
186  Radio.Sleep();
187 
188  int i;

190  loraTxtimes[i] = SX1276GetTimeOnAir(MODEM_LORA, LORA_BANDWIDTH,
191  LORA_SPREADING_FACTOR, LORA_CODINGRATE, LORA_PREAMBLE_LENGTH,
