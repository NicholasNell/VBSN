390 static bool processRXBuffer( ) {
391  memcpy(&rxDatagram, &RXBuffer, loraRxBufferSize);
392 
393  if (receivedMsgIndex == MAX_STORED_MSG) {
394  receivedMsgIndex = 0;
395  }
396  receivedDatagrams[receivedMsgIndex] = rxDatagram;
397 
398  // check if message was meant for this node:
399  if ((rxDatagram.msgHeader.nextHop == _nodeID)
400  || (rxDatagram.msgHeader.nextHop == BROADCAST_ADDRESS)) { // if destination is this node or broadcast check message type
401 
402  switch (rxDatagram.msgHeader.flags) {
403  case MSG_RTS: // RTS
404  MACSend(MSG_CTS, rxDatagram.msgHeader.localSource); // send CTS to requesting node
405  MACState = MAC_LISTEN;
406  break;
407  case MSG_CTS: // CTS
408  if (hasData) {
409  MACSend(MSG_DATA, rxDatagram.msgHeader.localSource); // Send data to destination node
410  MACState = MAC_LISTEN;
411  }
412  else {
413  MACState = MAC_SLEEP;
414  }
415  break;
416  case MSG_DATA: // DATA
417  if (rxDatagram.msgHeader.netDest != _nodeID) {
418  hopMessageFlag = true;
419  }
420  else {
421  hopMessageFlag = false;
422  // now either store the data for further later sending or get ready to upload data to cloud
423  }
424 
425  MACSend(MSG_ACK, rxDatagram.msgHeader.localSource); // Send Ack back to transmitting node
426  if (!hopMessageFlag) {
427  MACState = MAC_SLEEP;
428  }
429  else {
430  MACState = MAC_RTS;
431  }
432 
433  break;
434  case MSG_ACK: // ACK
435  hasData = false; // data has succesfully been sent
436  MACState = MAC_SLEEP;
437  break;
438  case MSG_RREP: // RREP
439  MACSend(MSG_ACK, rxDatagram.msgHeader.localSource); // Send Ack back to transmitting node
440  nextNetOp = processRrep();
441  MACState = MAC_NET_OP;
442  netOp = true;
443  break;
444 
445  /* SYNC and RREQ messages are broadcast, thus no ack, only RRep if route is known */
446  case MSG_SYNC: // SYNC
447  {
448  addNeighbour(
449  rxDatagram.msgHeader.localSource,
450  rxDatagram.msgHeader.txSlot);
451 
452  MACState = MAC_SLEEP;
453 
454  if (rxDatagram.msgHeader.localSource == _nodeID) { // change ID because another node has the same ID
455  genID(true);
456  schedChange = true;
457  }
458  addRoutetoNeighbour(rxDatagram.msgHeader.localSource);
459  break;
460  }
461  case MSG_RREQ: // RREQ
462  // when a rreq is received, determine what actions to do next.
463  nextNetOp = processRreq();
464  MACState = MAC_NET_OP;
465  netOp = true;
466  break;
467  default:
468  return false;
469  }
470  return true;
471  }
472  else {
473  MACState = MAC_SLEEP;
474  return false;
475  }
476 }

492  * uint8_t hops; // number of message hops from source to here

495  * uint8_t flags; // see flags
496  */
497  rxDatagram.msgHeader.nextHop = destination;

503 
504  }

510 
511  switch (msgType) {

513  rxDatagram.msgHeader.ttl = 1;
514  rxDatagram.msgHeader.netDest = BROADCAST_ADDRESS;
515  rxDatagram.msgHeader.nextHop = BROADCAST_ADDRESS;
516  break;

531  nodeAddress source_addr;
532  uint8_t source_sequence_num;
533  uint8_t broadcast_id;
534  nodeAddress dest_addr; // destination address: will probably be a gateway
