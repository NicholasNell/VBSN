166 void SystemClockConfig( void ) {
167  /*
168  * What clock goes where??:
169  * ACLK: use with TIMER_A and SPI
170  * SMCLK: use with SPI and TIMER_A
171  */
172  /* Configuring pins for peripheral/crystal usage */
173  MAP_GPIO_setAsPeripheralModuleFunctionOutputPin(GPIO_PORT_PJ,
174  GPIO_PIN0 | GPIO_PIN1, GPIO_PRIMARY_MODULE_FUNCTION);
175  CS_setExternalClockSourceFrequency(LFXT_FREQ, HFXT_FREQ); //LFXT_FREQ
176  /* Starting LFXT in non-bypass mode without a timeout. */
177  CS_startLFXT(CS_LFXT_DRIVE3);
178  MAP_CS_setReferenceOscillatorFrequency(REFO_FREQ);
179 
180  MAP_FPU_enableModule(); // makes sure the FPU is enabled before DCO frequency tuning
181  MAP_CS_setDCOFrequency(1500000);
182 
183  MAP_CS_initClockSignal(CS_ACLK, ACLK_SOURCE, ACLK_DIV); //128kHz
184  MAP_CS_initClockSignal(CS_MCLK, MCLK_SOURCE, MCLK_DIV); // 1.5MHz
185 // MAP_CS_initClockSignal(CS_HSMCLK, HSMCLK_SOURCE, HSMCLK_DIV); // 24MHz
186  MAP_CS_initClockSignal(CS_SMCLK, SMCLK_SOURCE, SMCLK_DIV);
187  MAP_CS_initClockSignal(CS_BCLK, CS_LFXTCLK_SELECT, CS_CLOCK_DIVIDER_1);
