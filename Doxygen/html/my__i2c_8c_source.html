52  I2C_setSlaveAddress(EUSCI_B1_BASE, addr); // Make sure correct i2c slave selected
53  I2C_setMode(EUSCI_B1_BASE, EUSCI_B_I2C_TRANSMIT_MODE); // set to transmit mode for configuration
54  while (MAP_I2C_masterIsStopSent(EUSCI_B1_BASE))
55  ;
56 
57  int index = 0;
58  retval = I2C_masterSendMultiByteStartWithTimeout(
59  EUSCI_B1_BASE,

61  10); // first byte is reg to be configured
62  int count = 0;
63  while (count < bufLen) {
64  retval = I2C_masterSendMultiByteNextWithTimeout(
65  EUSCI_B1_BASE,
