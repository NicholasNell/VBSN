218 static uint32_t SX1276GetLoRaTimeOnAirNumerator(uint32_t bandwidth,
219  uint32_t datarate, uint8_t coderate, uint16_t preambleLen, bool fixLen,
220  uint8_t payloadLen, bool crcOn) {
221  int32_t crDenom = coderate + 4;
222  bool lowDatareOptimize = false;
223 
224  // Ensure that the preamble length is at least 12 symbols when using SF5 or
225  // SF6
226  if ((datarate == 5) || (datarate == 6)) {
227  if (preambleLen < 12) {
228  preambleLen = 12;
229  }
230  }
231 
232  if (((bandwidth == 0) && ((datarate == 11) || (datarate == 12)))
233  || ((bandwidth == 1) && (datarate == 12))) {
234  lowDatareOptimize = true;
235  }
236 
237  int32_t ceilDenominator;
238  int32_t ceilNumerator = (payloadLen << 3) + (crcOn ? 16 : 0)
239  - (4 * datarate) + (fixLen ? 0 : 20);
240 
241  if (datarate <= 6) {
242  ceilDenominator = 4 * datarate;
243  } else {
244  ceilNumerator += 8;
245 
246  if (lowDatareOptimize == true) {
247  ceilDenominator = 4 * (datarate - 2);
248  } else {
249  ceilDenominator = 4 * datarate;
250  }
251  }
252 
253  if (ceilNumerator < 0) {
254  ceilNumerator = 0;
255  }
256 
257  // Perform integral ceil()
258  int32_t intermediate = ((ceilNumerator + ceilDenominator - 1)
259  / ceilDenominator) * crDenom + preambleLen + 12;
260 
261  if (datarate <= 6) {
262  intermediate += 2;
263  }
264 
265  return (uint32_t) ((4 * intermediate + 1) * (1 << (datarate - 2)));
266 }
