239 void SX1276SetRfTxPower( int8_t power ) {
240  uint8_t paConfig = 0;
241  uint8_t paDac = 0;
242 
243  paConfig = spiRead_RFM( REG_PACONFIG);
244  paDac = spiRead_RFM( REG_PADAC);
245 
246  paConfig = (paConfig & RF_PACONFIG_PASELECT_MASK)
247  | SX1276GetPaSelect(power);
248 
249  if ((paConfig & RF_PACONFIG_PASELECT_PABOOST)
250  == RF_PACONFIG_PASELECT_PABOOST) {
251  if (power > 17) {
252  paDac = (paDac & RF_PADAC_20DBM_MASK) | RF_PADAC_20DBM_ON;
253  }
254  else {
255  paDac = (paDac & RF_PADAC_20DBM_MASK) | RF_PADAC_20DBM_OFF;
256  }
257  if ((paDac & RF_PADAC_20DBM_ON) == RF_PADAC_20DBM_ON) {
258  if (power < 5) {
259  power = 5;
260  }
261  if (power > 20) {
262  power = 20;
263  }
264  paConfig = (paConfig & RF_PACONFIG_OUTPUTPOWER_MASK)
265  | (uint8_t) ((uint16_t) (power - 5) & 0x0F);
266  }
267  else {
268  if (power < 2) {
269  power = 2;
270  }
271  if (power > 17) {
272  power = 17;
273  }
274  paConfig = (paConfig & RF_PACONFIG_OUTPUTPOWER_MASK)
275  | (uint8_t) ((uint16_t) (power - 2) & 0x0F);
276  }
277  }
278  else {
279  if (power > 0) {
280  if (power > 15) {
281  power = 15;
282  }
283  paConfig = (paConfig & RF_PACONFIG_MAX_POWER_MASK
284  & RF_PACONFIG_OUTPUTPOWER_MASK) | (7 << 4) | (power);
285  }
286  else {
287  if (power < -4) {
288  power = -4;
289  }
290  paConfig = (paConfig & RF_PACONFIG_MAX_POWER_MASK
291  & RF_PACONFIG_OUTPUTPOWER_MASK) | (0 << 4) | (power + 4);
292  }
293  }
294  spiWrite_RFM( REG_PACONFIG, paConfig);
295  spiWrite_RFM( REG_PADAC, paDac);
296 }
